        def bearing(lat1, lon1, lat2, lon2):
            # Convert from degrees to radians
            lat1 = math.radians(lat1)
            lon1 = math.radians(lon1)
            lat2 = math.radians(lat2)
            lon2 = math.radians(lon2)

            dlon = lon2 - lon1
            x = math.sin(dlon) * math.cos(lat2)
            y = math.cos(lat1) * math.sin(lat2) - math.sin(lat1) * math.cos(lat2) * math.cos(dlon)
            initial_bearing = math.atan2(x, y)
            # Convert from radians to degrees and normalize to 0..360
            bearing_deg = (math.degrees(initial_bearing) + 360) % 360
            return bearing_deg
        def geo_sat_elevation(lat_sat, lon_sat, lat_tgt, lon_tgt, h_sat=35786000.0):
            # Earth radius in meters
            R = 6378137.0
            # Convert degrees to radians
            lat_sat = math.radians(lat_sat)
            lon_sat = math.radians(lon_sat)
            lat_tgt = math.radians(lat_tgt)
            lon_tgt = math.radians(lon_tgt)
            # Central angle (great-circle) between sat and target
            cos_c = (math.sin(lat_sat) * math.sin(lat_tgt) +
                     math.cos(lat_sat) * math.cos(lat_tgt) * math.cos(lon_tgt - lon_sat))
            c = math.acos(cos_c)
            # Law of cosines for the triangle (Earth center - sat - tgt)
            # S = satellite distance from Earth's center
            S = R + h_sat
            # elevation = angle at satellite between nadir and direction to target
            elevation = math.degrees(math.asin(R / S * math.sin(c)))
            return elevation - 90

            def split_and_move_single_file(args):
                src_folder, dst_folder, filename = args

                def is_tle_line(line):
                    return line.startswith('1 ') or line.startswith('2 ')

                src_path = os.path.join(src_folder, filename)
                with open(src_path, 'r') as f:
                    lines = [line.strip() for line in f if line.strip()]

                tle_sets = []
                i = 0
                while i + 2 < len(lines):
                    if is_tle_line(lines[i + 1]) and is_tle_line(lines[i + 2]):
                        tle_sets.append(lines[i:i + 3])
                        i += 3
                    else:
                        i += 1

                if len(tle_sets) == 1:
                    shutil.move(src_path, os.path.join(dst_folder, filename))
                    # print(f"Moved: {filename}")
                elif len(tle_sets) > 1:
                    for idx, tle in enumerate(tle_sets):
                        tle_name = tle[0].replace(' ', '_')[:24]
                        out_filename = f"{tle_name}_{idx + 1}.tle"
                        out_path = os.path.join(dst_folder, out_filename)
                        with open(out_path, 'w') as out_f:
                            out_f.write('\n'.join(tle) + '\n')
                        # print(f"Extracted: {out_filename}")
                else:
                    print(f"Warning: No valid TLEs in {filename}")

            def split_and_move_tle_files(src_folder, dst_folder, max_workers=multiprocessing.cpu_count()):
                """
                Splits multi-TLE files into individual files in dst_folder.
                Moves single-TLE files as-is to dst_folder.
                """
                os.makedirs(dst_folder, exist_ok=True)
                tle_files = [f for f in os.listdir(src_folder) if f.lower().endswith('.tle')]
                args_list = [(src_folder, dst_folder, f) for f in tle_files]
                with ThreadPoolExecutor(max_workers=max_workers) as executor:
                    executor.map(split_and_move_single_file, args_list)

            def mean_motion_to_sma(mean_motion):
                # Constante gravitationnelle terrestre (km^3/s^2)
                mu = 398600.4418
                # Mean motion (rev/day) → period (s)
                period = 86400 / mean_motion
                # SMA via Kepler's third law
                a = (mu * (period / (2 * pi)) ** 2) ** (1 / 3)
                return a

            def parse_tle_block(lines):
                # lines = [name, line1, line2]
                name = lines[0].strip()
                line1 = lines[1].strip()
                line2 = lines[2].strip()
                inc = float(line2[8:16])  # Inclinaison (deg)
                raan = float(line2[17:25])  # RAAN (deg)
                ecc = float("0." + line2[26:33])  # Excentricité
                argp = float(line2[34:42])  # Argument du périastre (deg)
                mean_anomaly = float(line2[43:51])  # Anomalie moyenne (deg)
                mean_motion = float(line2[52:63])  # Révolutions par jour
                sma = mean_motion_to_sma(mean_motion)
                return {
                    "name": name,
                    "inclination": round(inc, 4),
                    "raan": round(raan, 4),
                    "eccentricity": round(ecc, 7),
                    "argp": round(argp, 4),
                    "sma_km": round(sma, 2),
                    "tle_line1": line1,
                    "tle_line2": line2
                }

            def read_all_tle(folder=input_TLE):
                files = glob(os.path.join(folder, "*.tle"))
                all_tle = []
                for filename in files:
                    with open(filename) as f:
                        lines = f.readlines()
                        # chaque sat = 3 lignes
                        for i in range(0, len(lines), 3):
                            if i + 2 < len(lines):
                                tle_block = lines[i:i + 3]
                                tle_data = parse_tle_block(tle_block)
                                tle_data["filename"] = filename
                                all_tle.append(tle_data)
                return all_tle

            # Lecture de tous les TLE
            tle_list = read_all_tle(Path(self.data_dir).expanduser()/'processed_single_TLE')
            # Création DataFrame
            df = pd.DataFrame(tle_list)
            df.sort_values("raan", ascending=True, inplace=True)
            df.reset_index(drop=True, inplace=True)
            def pick_satellites_by_raan(df, n):
                if n == -1:
                    return df
                import numpy as np  # Just in case
                if df.shape[0] < n:
                    n = df.shape[0]
                raan_targets = np.linspace(0, 360, n, endpoint=False)
                picked_indices = []
                used_indices = set()

                for target in raan_targets:
                    remaining = df.index.difference(used_indices)
                    idx = df.loc[remaining, 'raan'].sub(target).abs().idxmin()
                    picked_indices.append(idx)
                    used_indices.add(idx)
                # Don't forget to return the selected satellites!
                return df.loc[picked_indices].reset_index(drop=True)

            selected_df = pick_satellites_by_raan(df, n)
            self.selected_df = selected_df
            ### DISTRIBUTIONS
            work_plan_metadata = []
            workers_planes_dist = []
            for i in range(selected_df.shape[0]):
                work_plan_metadata.append((i, 1))
                workers_planes_dist.append([i])
            workers_planes_dist = [workers_planes_dist]


            work_plan_metadata = WorkDispatcher.make_chunks(len(work_plan_metadata),
                work_plan_metadata, verbose=self.verbose, workers=workers,
                threshold=12)
            workers_link_dist = []
            for worker in work_plan_metadata:
                worker_list = []
                for chunk in worker:
                    worker_list.append([chunk[0]])
                workers_link_dist.append(worker_list)
        except Exception as e:
            print(traceback.format_exc())
            print(f'warning issue while trying to build distribution: {e}')
        return workers_link_dist, work_plan_metadata, 'plane', 'files', 'ko'

agi_env API

This is a module share between core/agi-core and gui/agilab to set all the path required by both

Instanciation

from agi_env import AgiEnv

env = AgiEnv('my-code')

# or
path_to_ = Path("../../agi/apps/my-code-project/src//.py")
env = AgiEnv(path_to_)
